        } catch (e) {

    // Function to load editor content from localStorage
    function loadEditorContent() {
        try {
            const savedContent = localStorage.getItem('savedEditorContent');
            const timestamp = localStorage.getItem('editorContentTimestamp');
            const savedLang = localStorage.getItem('savedEditorLanguage');
            
            // Check if we have saved content and it's not too old (24 hour expiry)
            if (savedContent && timestamp) {
                const currentTime = new Date().getTime();
                const hoursSinceSaved = (currentTime - parseInt(timestamp)) / (1000 * 60 * 60);
                
                // If content is too old (more than 24 hours), don't restore
                if (hoursSinceSaved > 24) {
                    localStorage.removeItem('savedEditorContent');
                    localStorage.removeItem('editorContentTimestamp');
                    localStorage.removeItem('savedEditorLanguage');
                    return false;
                }
                
                const content = JSON.parse(savedContent);
                
                // Only restore if there's actual content
                if (content && content.length > 0) {
                    
                    // First restore the language if available
                    if (savedLang) {
                        lang = savedLang;
                        $('#lang-select').dropdown('set selected', lang);
                        editor.session.setMode("ace/mode/" + get_mode(lang));
                    }
                    
                    // Clear existing tabs except the first one
                    const tabsToRemove = $(".nav-tabs").children('li').not(':first').not(':last');
                    tabsToRemove.remove();
                    
                    // Reset counters
                    editor_cnt = 1;
                    editor_index = 1;
                    
                    let activeIndex = 0;
                    
                    // Create all tabs and set content
                    for (let i = 0; i < content.length; i++) {
                        if (content[i].active === 1) {
                            activeIndex = i;
                        }
                        
                        if (i === 0) {
                            // Update the first tab
                            $("#editor-1").children('a').html(content[i].file_name);
                            editor_session[0].setValue(content[i].code);
                        } else {
                            // Create new tabs
                            editor_cnt += 1;
                            editor_index += 1;
                            $('.add-editor').closest('li').before('<li id="editor-' + editor_cnt + '"><a data-toggle="tab">' + content[i].file_name + '</a> <span> <i class="fa fa-times"></i></span></li>');
                            
                            editor_session[i] = ace.createEditSession('', "ace/mode/" + get_mode(lang));
                            editor_session[i].setValue(content[i].code);
                        }
                    }
                    
                    // Set the active tab
                    editor.setSession(editor_session[activeIndex]);
                    $("#editor-" + (activeIndex + 1)).children('a').tab('show');
                    active_editor_id = $("#editor-" + (activeIndex + 1)).children('a');
                    active_file_name = active_editor_id.html();
                    
                    // Show notification to user
                    // toastr.info('Your previous work has been restored', 'Welcome back!');
                    
                    return true;
                }
            }
        } catch (e) {
            console.error('Error loading editor content:', e);
        }
        
        return false;
    }
